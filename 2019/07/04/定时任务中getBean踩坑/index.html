定时任务中getBean踩坑
日常工作
Java
发布日期:   2019-07-03
阅读次数:  

2019-07-03
在进行开发时，需要在定时任务中调用service层进行任务处理。因为TimerTask类并没有兼容Spring框架，所以无法使用Autowired进行注入。因为注入的话，@Autowired时候报空指针。所以只能使用最原始的方法，通过getbean注入，拿到service层。具体代码如下:
1.SpringContextUtil类代码：

public class SpringContextUtil implements ApplicationContextAware {

private static ApplicationContext applicationContext; // Spring应用上下文环境

/**
 * 实现ApplicationContextAware接口的回调方法，设置上下文环境
 * 
 * @param applicationContext
 * @throws BeansException
 */
public void setApplicationContext(ApplicationContext applicationContext) throws BeansException {
    SpringContextUtil.applicationContext = applicationContext;
}

/**
 * @return ApplicationContext
 */
public static ApplicationContext getApplicationContext() {
    return applicationContext;
}

/**
 * 获取对象
 * 
 * @param name
 * @return Object 一个以所给名字注册的bean的实例
 * @throws BeansException
 */
public static Object getBean(String name) throws BeansException {
    return applicationContext.getBean(name);
}

/**
 * 获取类型为requiredType的对象
 * 如果bean不能被类型转换，相应的异常将会被抛出（BeanNotOfRequiredTypeException）
 * 
 * @param name
 *            bean注册名
 * @param requiredType
 *            返回对象类型
 * @return Object 返回requiredType类型对象
 * @throws BeansException
 */
@SuppressWarnings({ "unchecked", "rawtypes" })
public static Object getBean(String name, Class requiredType) throws BeansException {
    return applicationContext.getBean(name, requiredType);
}

/**
 * 如果BeanFactory包含一个与所给名称匹配的bean定义，则返回true
 * 
 * @param name
 * @return boolean
 */
public static boolean containsBean(String name) {
    return applicationContext.containsBean(name);
}

/**
 * 判断以给定名字注册的bean定义是一个singleton还是一个prototype。
 * 如果与给定名字相应的bean定义没有被找到，将会抛出一个异常（NoSuchBeanDefinitionException）
 * 
 * @param name
 * @return boolean
 * @throws NoSuchBeanDefinitionException
 */
public static boolean isSingleton(String name) throws NoSuchBeanDefinitionException {
    return applicationContext.isSingleton(name);
}

/**
 * @param name
 * @return Class 注册对象的类型
 * @throws NoSuchBeanDefinitionException
 */
@SuppressWarnings("rawtypes")
public static Class getType(String name) throws NoSuchBeanDefinitionException {
    return applicationContext.getType(name);
}

/**
 * 如果给定的bean名字在bean定义中有别名，则返回这些别名
 * 
 * @param name
 * @return
 * @throws NoSuchBeanDefinitionException
 */
public static String[] getAliases(String name) throws NoSuchBeanDefinitionException {
    return applicationContext.getAliases(name);
}

}

2.业务中的设备更新任务:DeviceUpdateTimerTask
这个类继承了TimeTask，重写了run方法，也就把业务写在这个方法里，代码如下:
public class DeviceUpdateTimerTask extends TimerTask {
public static final Logger logger = LoggerFactory.getLogger(DeviceUpdateTimerTask.class);
public static final int Visitor_Expire = 0;
public static final int Visitor_Valid = 1;
public static final int Human_InActive = 0;
public static final int Human_Active = 1;
public static final int Human_Prepare = 2;

@Override
public void run() {
    VisitorService visitorService = (VisitorService) SpringContextUtil.getBean("visitorServiceImpl");
    CloudSideClientManager cloudSideClientManager = (CloudSideClientManager) SpringContextUtil
            .getBean("cloudSideClientManager");
    HumanService humanService = (HumanService) SpringContextUtil.getBean("humanServiceImp");
    DeviceService deviceService = (DeviceService) SpringContextUtil.getBean("deviceServiceImp");


    try {
        // 这个判断是用老方法@Autowired注入的时候 报空指针 测试的时候在这儿判断了一下 是因为service空 没有成功注入
        // 所有service/dao注入需要SpringContextUtil.getBean才可以
        if (visitorService == null) {
            logger.error(">>>>ERROR:visitorService为null,无法更新>>>>");
            throw new NotFoundException("VisitorService为null,无法执行访客状态定时更新任务");
        }
        if (humanService == null) {
            logger.error(">>>>ERROR:humanService为null,无法更新>>>>");
            throw new NotFoundException("HumanService为null,无法执行人员状态定时更新任务");
        }
        if (cloudSideClientManager == null) {
            logger.error(">>>>ERROR:cloudSideClientManager为null,无法更新>>>>");
            throw new NotFoundException("CloudSideClientManager为null,无法执行设备更新任务");
        }
        if (deviceService == null) {
            logger.error(">>>>ERROR:deviceService为null,无法更新>>>>");
            throw new NotFoundException("DeviceService为null,无法获取设备序列号");
        }

        // 1.更新过期访客
        Map<String, Object> map = new HashMap<>();
        map.put("visitStatus", Visitor_Expire);
        boolean boo = visitorService.updateExpireVisitor(map);
        logger.info(">>>>INFO:访客状态更新>>>>:" + (boo ? "成功" : "失败"));
        // 2.修改人员状态(humanStatus为2并且开通时间小于当前时间)修改状态为1
        Map<String, Object> mp = new HashMap<>();
        mp.put("humanStatus", Human_Active);
        boolean bool = humanService.updatePrepareHuman(mp);
        logger.info(">>>>INFO:人员状态更新>>>>:" + (bool ? "成功" : "失败"));
        // 3.下发指令让所有设备更新
        //查询所有在线的设备--getOnlineDeviceList
        List<String> serialList = deviceService.getOnlineDeviceAll();
        for (String serial : serialList) {
            cloudSideClientManager.clientRegister(serial,"127.0.0.1", 50053);
            // 2.调用通讯接口让区域内设备更新
            CloudSideClientProxy ccp =cloudSideClientManager.getClient(serial);
            List<Person> personList = deviceService.getPermissionPersonByDevice(serial);
            if (ccp == null) {
                logger.error(">>>>ERROR:序列号为"+serial+"的设备未上线!>>>>");
                throw new DeviceOfflineException("序列号为"+serial+"的设备未上线!");
            }
            ccp.permissionUpdateAll(personList);
        }
    } catch (Exception e) {
        logger.error(">>>>ERROR:解析信息发生异常>>>>");
    }
}

}
